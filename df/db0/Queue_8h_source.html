60  bool Dispatch()
61  {
62  // Check if queue is empty
63  uint32_t length = queue_length;
64  if (length == 0)
65  return true;
66 
67  // Decrement queue length and check if we have another entry to run
68  queue_length = --length;
69  if (length != 0)
70  {
71  // Call next entry in queue
72  uint32_t head = queue_head;
73  QueueEntry *entry = &queue[head];
74  queue_head = (head + 1) % N;
75 
76  entry->func(entry->args);
77  return false;
78  }
79  return true;
80  }

85  void Enqueue(FT func, const T &args)
86  {
87  OS::DisableIRQ();
88 
89  uint32_t length = queue_length;
90  if (length == 0)
91  {
92  queue_length = 1;
93  func(args);
94  OS::EnableIRQ();
95  return;
96  }
97  if (length >= N)
98  {
99  OS::EnableIRQ();
100  if (W)
101  while (length >= N) INLINE_ASM("");
102  else
103  ExScreen::Abort("Queue overflow\n");
104  return;
105  }
106 
107  uint32_t tail = queue_tail;
108  queue_tail = (tail + 1) % N;
109  queue_length = length + 1;
110 
111  QueueEntry *entry = &queue[tail];
112  entry->func = func;
113  entry->args = args;
114 
115  OS::EnableIRQ();
116  }
