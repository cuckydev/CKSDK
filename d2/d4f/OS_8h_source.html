252  template<typename R, typename... A>
253  class Function
254  {
256  private:
257  typedef R (*F)(A...);
258  F ptr;
259 
260  public:
261  Function() { ptr = nullptr; }
262  Function(F _ptr) { ptr = _ptr; }
263  Function(Function &o) { ptr = o.ptr; }
264 
265  inline R operator()(A... arg)
266  {
267  // T9 must be set before calling the function
268  // as it's used for local addressing for DLLs
269  // It has no effect for boot functions
270  INLINE_ASM("move $t9, %0;" :: "r"(ptr) : "$t9");
271  return ptr(arg...);
272  }
273 
274  bool operator==(F _ptr) { return ptr == _ptr; }
275  bool operator==(Function &o) { return ptr == o.ptr; }
276  bool operator!=(F _ptr) { return ptr != _ptr; }
277  bool operator!=(Function &o) { return ptr != o.ptr; }
279  };
