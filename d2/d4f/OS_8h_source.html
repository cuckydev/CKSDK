376  template<typename R, typename... A>
377  class Function
378  {
380  private:
381  typedef R (*F)(A...);
382  F ptr;
383 
384  public:
385  Function() { ptr = nullptr; }
386  Function(F _ptr) { ptr = _ptr; }
387  Function(const Function &o) { ptr = o.ptr; }
388  void operator= (F _ptr) { ptr = _ptr; }
389  void operator= (const Function &o) { ptr = o.ptr; }
390 
391  inline R operator()(A... arg) const
392  {
393  // T9 must be set before calling the function
394  // as it's used for local addressing for DLLs
395  // It has no effect for boot functions
396  INLINE_ASM("move $t9, %0;" :: "r"(ptr) : "$t9");
397  return ptr(arg...);
398  }
399 
400  bool operator==(F _ptr) const { return ptr == _ptr; }
401  bool operator==(const Function &o) const { return ptr == o.ptr; }
402  bool operator!=(F _ptr) const { return ptr != _ptr; }
403  bool operator!=(const Function &o) const { return ptr != o.ptr; }
405  };
