339  template<typename R, typename... A>
340  class Function
341  {
343  private:
344  typedef R (*F)(A...);
345  F ptr;
346 
347  public:
348  Function() { ptr = nullptr; }
349  Function(F _ptr) { ptr = _ptr; }
350  Function(Function &o) { ptr = o.ptr; }
351 
352  inline R operator()(A... arg)
353  {
354  // T9 must be set before calling the function
355  // as it's used for local addressing for DLLs
356  // It has no effect for boot functions
357  INLINE_ASM("move $t9, %0;" :: "r"(ptr) : "$t9");
358  return ptr(arg...);
359  }
360 
361  bool operator==(F _ptr) { return ptr == _ptr; }
362  bool operator==(Function &o) { return ptr == o.ptr; }
363  bool operator!=(F _ptr) { return ptr != _ptr; }
364  bool operator!=(Function &o) { return ptr != o.ptr; }
366  };
