774  inline void Queue_ImageLoad(const void *addr, uint16_t x, uint16_t y, uint16_t w, uint16_t h)
775  {
776  uint32_t bcr = w * h;
777  // if (bcr & 1)
778  // ExScreen::Abort("LoadImage w * h is odd");
779  bcr >>= 1;
780 
781  uint32_t bs = 1;
782  while (((bcr & 1) == 0) && (bs < 16))
783  {
784  bs <<= 1;
785  bcr >>= 1;
786  }
787 
788  // if (bcr >= 65536)
789  // ExScreen::Abort("LoadImage bcr too large");
790  bcr <<= 16;
791  bcr |= bs;
792  Queue_ImageDMA(addr, (x << 0) | (y << 16), (w << 0) | (h << 16), bcr);
793  }
