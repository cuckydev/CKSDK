655  inline void LoadImage(const void *addr, uint16_t x, uint16_t y, uint16_t w, uint16_t h)
656  {
657  uint32_t bcr = w * h;
658  // if (bcr & 1)
659  // ExScreen::Abort("LoadImage w * h is odd");
660  bcr >>= 1;
661 
662  uint32_t bs = 1;
663  while (((bcr & 1) == 0) && (bs < 16))
664  {
665  bs <<= 1;
666  bcr >>= 1;
667  }
668 
669  // if (bcr >= 65536)
670  // ExScreen::Abort("LoadImage bcr too large");
671  bcr <<= 16;
672  bcr |= bs;
673  DMAImage(addr, (x << 0) | (y << 16), (w << 0) | (h << 16), bcr);
674  }
