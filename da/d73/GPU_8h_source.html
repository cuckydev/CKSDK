698  inline void LoadImage(const void *addr, uint16_t x, uint16_t y, uint16_t w, uint16_t h)
699  {
700  uint32_t bcr = w * h;
701  // if (bcr & 1)
702  // ExScreen::Abort("LoadImage w * h is odd");
703  bcr >>= 1;
704 
705  uint32_t bs = 1;
706  while (((bcr & 1) == 0) && (bs < 16))
707  {
708  bs <<= 1;
709  bcr >>= 1;
710  }
711 
712  // if (bcr >= 65536)
713  // ExScreen::Abort("LoadImage bcr too large");
714  bcr <<= 16;
715  bcr |= bs;
716  DMAImage(addr, (x << 0) | (y << 16), (w << 0) | (h << 16), bcr);
717  }
