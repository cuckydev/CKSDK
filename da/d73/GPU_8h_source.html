531  inline void LoadImage(const void *addr, uint16_t x, uint16_t y, uint16_t w, uint16_t h)
532  {
533  uint32_t bcr = w * h;
534  // if (bcr & 1)
535  // ExScreen::Abort("LoadImage w * h is odd");
536  bcr >>= 1;
537 
538  uint32_t bs = 1;
539  while (((bcr & 1) == 0) && (bs < 16))
540  {
541  bs <<= 1;
542  bcr >>= 1;
543  }
544 
545  // if (bcr >= 65536)
546  // ExScreen::Abort("LoadImage bcr too large");
547  bcr <<= 16;
548  bcr |= bs;
549  DMAImage(addr, (x << 0) | (y << 16), (w << 0) | (h << 16), bcr);
550  }
