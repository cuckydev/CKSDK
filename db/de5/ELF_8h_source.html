35  inline uint32_t ElfHash(const char *str)
36  {
37  uint32_t value = 0;
38  while (*str != '\0')
39  {
40  value <<= 4;
41  value += uint32_t(*(str++));
42 
43  uint32_t nibble = value & 0xF0000000;
44  if (nibble != 0)
45  value ^= nibble >> 24;
46 
47  value &= ~nibble;
48  }
49  return value;
50  }
